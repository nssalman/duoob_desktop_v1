class HomePage1 extends StatefulWidget {
  HomePage1({super.key});

  @override
  State<HomePage1> createState() => _HomePage1State();
}

class _HomePage1State extends State<HomePage1>
    with SingleTickerProviderStateMixin {
  final TaskController taskController = Get.put(TaskController());
  late TabController _tabController;

  final List<String> tabs = [
    'ERP Tasks',
    'Employee Tasks',
  ];

  @override
  void initState() {
    super.initState();
    _tabController = TabController(length: tabs.length, vsync: this);
  }

  @override
  void dispose() {
    _tabController.dispose();
    super.dispose();
  }

  @override
  Widget build(BuildContext context) {
    WidgetsBinding.instance
        .addPostFrameCallback((_) => taskController.refreshTask());
    return GetBuilder<TaskController>(
      builder: (control) {
        return Padding(
          padding: const EdgeInsets.symmetric(horizontal: 16, vertical: 8),
          child: Column(children: [
            // gapH8,
            _buildGlassmorphicTabBar(),
            gapH8,
            Expanded(
              child: Stack(
                children: [
                  TabBarView(
                    controller: _tabController,
                    physics: const NeverScrollableScrollPhysics(),
                    children: [
                      Column(
                        children: [
                          Container(
                              // height: 30,
                              // width: MediaQuery.of(context).size.width * 0.96,
                              decoration: BoxDecoration(
                                color: Colors.grey.shade200,
                                // border: Border.all(color: Colors.grey),
                                borderRadius: BorderRadius.circular(10),
                              ),
                              alignment: Alignment.centerLeft,
                              padding: const EdgeInsets.fromLTRB(10, 5, 10, 5),
                              child: Row(
                                crossAxisAlignment: CrossAxisAlignment.center,
                                children: [
                                  Expanded(
                                    child: Text(
                                      'Select All',
                                      style: const TextStyle(
                                          color: Colors.black,
                                          fontSize: 14,
                                          fontWeight: FontWeight.bold),
                                    ),
                                  ),
                                  InkWell(
                                    onTap: control.isAllSelect
                                        ? control.clearSelection
                                        : control.selectAll,
                                    child: Icon(
                                      Icons.select_all,
                                      color: Colors.black,
                                    ),
                                  ),
                                ],
                              )),
                          SizedBox(
                            height: 10,
                          ),
                          Expanded(
                            child: control.isDD365Loading
                                ? const ShimmerLoader(
                                    title: 'ERP Task',
                                  )
                                : control.d365TaskList.isEmpty
                                    ? NoDataWarning(
                                        message: 'No ERP Task found',
                                        onRefresh: () => control.refreshTask(),
                                      )
                                    : FadeInLeft(
                                        child: RefreshIndicator(
                                          onRefresh: () =>
                                              control.refreshTask(),
                                          child: ListView.builder(
                                            padding: EdgeInsets.only(
                                                bottom:
                                                    control.checkForSelection()
                                                        ? 20
                                                        : 120),
                                            itemBuilder: (context, index) {
                                              return D365tasktile(
                                                task:
                                                    control.d365TaskList[index],
                                                onChanged: (value) {
                                                  control
                                                      .toggleSelection(index);
                                                },
                                                isSelected: control
                                                    .d365TaskList[index]
                                                    .isSelected,
                                                onTapLink: () async {
                                                  log('https://rakp.rpsmart.com/d365emailnotifications/D365FnOWorkflowsubmission.aspx?NotificationID=${control.d365TaskList[index].notificationId}');
                                                  final result =
                                                      await Navigator.push(
                                                    context,
                                                    MaterialPageRoute(
                                                      builder: (BuildContext
                                                              context) =>
                                                          TaskWebViewScreen(
                                                              url:
                                                                  // 'https://webbrowsertools.com/test-download-with/'
                                                                  'https://rakp.rpsmart.com/d365emailnotifications/D365FnOWorkflowsubmission.aspx?NotificationID=${control.d365TaskList[index].notificationId}'),
                                                    ),
                                                  );
                                                  if (result == true) {
                                                    control.refreshTask()();
                                                  }
                                                },
                                              );
                                            },
                                            itemCount:
                                                control.d365TaskList.length,
                                          ),
                                        ),
                                      ),
                          ),
                          if (control.checkForSelection())
                            Padding(
                              padding: const EdgeInsets.only(
                                  left: 15,
                                  right: 15,
                                  bottom: kBottomNavigationBarHeight +
                                      kToolbarHeight),
                              child: Row(
                                children: [
                                  Expanded(
                                    child: _buildActionButton(control, context,
                                        isApprove: true),
                                  ),
                                  SizedBox(
                                    width: 10,
                                  ),
                                  Expanded(
                                    child: _buildActionButton(control, context),
                                  ),
                                ],
                              ),
                            )
                        ],
                      ),
                      Column(children: [
                        Expanded(
                          child: control.isTaskListLoading
                              ? const ShimmerLoader(
                                  title: 'Employee Task',
                                )
                              : control.userTaskMap.isEmpty
                                  ? NoDataWarning(
                                      message: 'No Employee Task found',
                                      onRefresh: () => control.refreshTask(),
                                    )
                                  : RefreshIndicator(
                                      onRefresh: () => control.refreshTask(),
                                      child: ListView.builder(
                                        padding: EdgeInsets.only(bottom: 120),
                                        itemCount: control.userTaskMap.length,
                                        itemBuilder: (context, index) {
                                          final user = control.userTaskMap.keys
                                              .elementAt(index);
                                          final tasks =
                                              control.userTaskMap[user]!;

                                          return Column(
                                            crossAxisAlignment:
                                                CrossAxisAlignment.start,
                                            children: [
                                              if (tasks.isNotEmpty)
                                                Container(
                                                    height: 30,
                                                    decoration: BoxDecoration(
                                                      color:
                                                          Colors.grey.shade200,
                                                      // border: Border.all(color: Colors.grey),
                                                      borderRadius:
                                                          BorderRadius.circular(
                                                              10),
                                                    ),
                                                    alignment:
                                                        Alignment.centerLeft,
                                                    padding: const EdgeInsets
                                                        .fromLTRB(10, 5, 10, 5),
                                                    child: Row(
                                                      children: [
                                                        Expanded(
                                                          child: Text(
                                                            'Tasks of ${user.taskOf ?? ''}',
                                                            style: const TextStyle(
                                                                color: Colors
                                                                    .black,
                                                                fontSize: 14,
                                                                fontWeight:
                                                                    FontWeight
                                                                        .bold),
                                                          ),
                                                        ),
                                                      ],
                                                    )),
                                              ...tasks.map((task) => TaskTile(
                                                    task: task,
                                                    onPressed: () async {
                                                      String? redirectLink =
                                                          await control
                                                              .redirectToWeb(
                                                                  task);
                                                      if (redirectLink !=
                                                          null) {
                                                        if (Platform.isIOS) {
                                                          final result =
                                                              await Navigator
                                                                  .push(
                                                            context,
                                                            MaterialPageRoute(
                                                              builder: (BuildContext
                                                                      context) =>
                                                                  TaskWebViewScreen(
                                                                      url: redirectLink
                                                                          .toString()),
                                                            ),
                                                          );
                                                          if (result != null &&
                                                              result == true) {
                                                            control
                                                                .refreshTask();
                                                          }
                                                        } else {
                                                          final result =
                                                              await Navigator
                                                                  .push(
                                                            context,
                                                            MaterialPageRoute(
                                                              builder: (BuildContext
                                                                      context) =>
                                                                  NativeWebViewWidget(
                                                                url: redirectLink
                                                                    .toString(),
                                                              ),
                                                            ),
                                                          );
                                                          if (result != null &&
                                                              result == true) {
                                                            control
                                                                .refreshTask();
                                                          }
                                                          // if (await canLaunchUrl(
                                                          //     Uri.parse(
                                                          //         redirectLink
                                                          //             .toString()))) {
                                                          //   await launchUrl(
                                                          //     Uri.parse(
                                                          //         redirectLink
                                                          //             .toString()),
                                                          //     mode: LaunchMode
                                                          //         .externalApplication,
                                                          //   );
                                                          // }
                                                        }
                                                      }
                                                    },
                                                  )),
                                            ],
                                          );
                                        },
                                      ),
                                    ),
                        )
                      ])
                    ],
                  ),
                ],
              ),
            ),
          ]),
        );
      },
    );
  }
